Joystic -> RAMPS 1.4
    Red (Vcc) -> AUX(4) 5V pin
    Orange (X-AXON)-> T0 A13
    Brown (y-AXON) -> T2 A15
    Green (switch) -> AUX(4) D32
    Blue (GND)-> AUC(4) GND

Enstops -> RAMPS 1.4

    Endstp1 -> D3 - (-) 
    Endstp2 -> D2 - (-)
    Endstp3 -> D14 - (-)
    Endstp4 -> D15 - (-)
    Endstp5 -> D18 - (-)
    Endstp6 -> D19 - (-)
